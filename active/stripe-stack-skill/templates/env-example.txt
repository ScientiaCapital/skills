# -----------------------------------------------------------------------------
# Required: Core Stripe Keys
# -----------------------------------------------------------------------------

# Server-side secret key (NEVER expose to client)
# Get from: https://dashboard.stripe.com/apikeys
# Test mode: starts with sk_test_
# Live mode: starts with sk_live_
STRIPE_SECRET_KEY=sk_test_your_key_here

# Webhook signing secret (from Stripe Dashboard or CLI)
# Dashboard: Webhooks → Your endpoint → Signing secret
# CLI: Output from `stripe listen --forward-to localhost:3000/api/stripe/webhook`
STRIPE_WEBHOOK_SECRET=whsec_your_secret_here

# Client-side publishable key (safe to expose)
# Used for Stripe.js and checkout redirects
NEXT_PUBLIC_STRIPE_PUBLISHABLE_KEY=pk_test_your_key_here

# =============================================================================
# Setup Instructions
# =============================================================================
#
# LOCAL DEVELOPMENT:
# 1. Copy this file to .env.local
# 2. Replace placeholder values with your test mode keys
# 3. For webhooks, run: stripe listen --forward-to localhost:3000/api/stripe/webhook
# 4. Copy the CLI-provided webhook secret to STRIPE_WEBHOOK_SECRET
#
# VERCEL DEPLOYMENT:
# 1. Go to Project → Settings → Environment Variables
# 2. Add each variable for Production environment
# 3. For live mode, use sk_live_ and pk_live_ keys
# 4. Create a new webhook endpoint in Stripe Dashboard pointing to your Vercel URL
# 5. Use the new webhook's signing secret
#
# SHARED STRIPE ACCOUNT:
# If multiple projects share one Stripe account:
# - Same API keys across projects
# - DIFFERENT webhook secrets (each project has its own endpoint)
# - Price IDs can be shared or project-specific
#
